dist_normal = tfd.Normal(loc=[-1., 1], scale=[0.1, 0.5])



# Make independent distribution from a 2-batch Normal.
ind_orig = tfd.Independent(
    distribution= dist_normal)

print('ind_orig batch_shape', ind_orig.batch_shape, ' event_shape ', ind_orig.event_shape)

ind_new_dims = tfd.Independent(
    distribution= dist_normal,
    reinterpreted_batch_ndims=1)

# All batch dims have been 'absorbed' into event dims.

print('ind_new_dims batch_shape', ind_new_dims.batch_shape, ' event_shape ', ind_new_dims.event_shape)


# Make independent distribution from a 2-batch bivariate Normal.
ind = tfd.Independent(
    distribution=tfd.MultivariateNormalDiag(
        loc=[[-1., 1], [1, -1]],
        scale_identity_multiplier=[1., 0.5]),
    reinterpreted_batch_ndims=1)

# All batch dims have been 'absorbed' into event dims.
ind.batch_shape  # ==> []
ind.event_shape  # ==> [2, 2]
